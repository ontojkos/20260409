Функция ПолучитьСписокПринтеров()

	prn =  принтеры.EnumPrinterConnections();
	i = 0;
	Пока i < prn.Count()-1 Цикл
		Если prn.Item(i+1) = "Этикетка" и НЕ ЗначениеЗаполнено(ПринтерЭтикетка) Тогда

			ПринтерЭтикетка=prn.Item(i+1);
			//Прервать; 
		ИначеЕсли prn.Item(i+1) = "Отчет" и НЕ ЗначениеЗаполнено(ПринтерОтчет)Тогда
			//\\pos-pc\Argox OS-214 plus series PPLB
			//ВыборПринтеров.Добавить(prn.Item(i+1));
			ПринтерОтчет=prn.Item(i+1);
			//Прервать;	
		КонецЕсли;
		ВыборПринтеров.Добавить(prn.Item(i+1));
		i = i + 2;
	КонецЦикла;
	
	Возврат ВыборПринтеров;

Функция ГенерацияШтрихкода()
	
	//ЧастьШКАртикул = "0" + Прав(Объект.ССылкаНОМ.Артикул,5);
	ЧастьШКАртикул = "0" + Прав(Объект.ССылкаНОМГП.Код,5);
	
	ФорматШКВес = Формат(Вес,"ЧЦ=4; ЧДЦ=2; ЧРД=,; ЧВН=");
	ЧасьШКВес   = Лев(ФорматШКВес,2)+Прав(ФорматШКВес,2);
	
	ФорматШКДлина = Формат(Длина,"ЧЦ=5; ЧДЦ=2; ЧРД=,; ЧВН=");
	ЧасьШКДлина   = Лев(ФорматШКДлина,3)+Прав(ФорматШКДлина,2);
	
	ФорматШКДата      = Формат(ТекущаяДатаСеанса(),"ДФ=dd.MM.yy");
	ФорматШКДатаДень  = Лев(ФорматШКДата,2);
	ФорматШКДатаМесяц = Прав(Лев(ФорматШКДата,5),2);
	ФорматШКДатаГод   = Прав(ФорматШКДата,2);
	ЧасьШКДата        = ФорматШКДатаДень + ФорматШКДатаМесяц + ФорматШКДатаГод;
	
	ФорматШКВремя      = Формат(ТекущаяДатаСеанса(),"ДФ=HH.mm"); 
	ФорматШКДатаЧасы   = Лев(ФорматШКВремя,2);
	ФорматШКДатаМинуты = Прав(ФорматШКВремя,2);
	ЧасьШКВремя        = ФорматШКДатаЧасы + ФорматШКДатаМинуты;	
	
	ТекШтрихкод = ЧастьШКАртикул + ЧасьШКВес + ЧасьШКДлина + ЧасьШКДата + ЧасьШКВремя; 
	
	Возврат ТекШтрихкод;

		Возврат;
	КонецПопытки;
	
	ОборотыИзФайла = 0;
	ВесГраммИзФайла = 0;
	ИмяУстройства = "";
	ДатаОтчетаСтрокой = "";
	
	НайденыОбороты = Ложь;
	НайденВес = Ложь;
	
	Для НомерСтроки = 1 По ТекстДок.КоличествоСтрок() Цикл
		
		СтрокаФайла = ТекстДок.ПолучитьСтроку(НомерСтроки);
		СтрокаФайла = СтрЗаменить(СтрокаФайла, Символы.ВК, "");
		СтрокаФайла = СтрЗаменить(СтрокаФайла, Символы.ПС, "");
		СтрокаБезПробелов = СокрЛП(СтрокаФайла);
		
		Если НЕ ЗначениеЗаполнено(ДатаОтчетаСтрокой) Тогда
			ДатаОтчетаСтрокой = ИзвлечьДатуОтчетаLGLИзПервойСтроки(СтрокаФайла);
		КонецЕсли;
		
		Если НЕ ЗначениеЗаполнено(ИмяУстройства) Тогда
			ИмяУстройства = ИзвлечьИдентификаторСтанкаLGL(СтрокаФайла);
		КонецЕсли;
		
		Если НЕ НайденыОбороты И Лев(СтрокаБезПробелов, 11) = "Revolution," Тогда
			Части = СтрРазделить(СтрокаБезПробелов, ",");
			Если Части.Количество() >= 2 Тогда
				Попытка
					ОборотыИзФайла = Число(СтрЗаменить(СокрЛП(Части[1]), ".", ","));
					НайденыОбороты = Истина;
				Исключение
				КонецПопытки;
			КонецЕсли;
		КонецЕсли;
		
		Если НЕ НайденВес И Лев(СтрокаБезПробелов, 12) = "Consumption," Тогда
			Части = СтрРазделить(СтрокаБезПробелов, ",");
			Если Части.Количество() >= 3 Тогда
				Если СокрЛП(Части[2]) = "g" Тогда
					Попытка
						ВесГраммИзФайла = Число(СтрЗаменить(СокрЛП(Части[1]), ".", ","));
						НайденВес = Истина;
					Исключение
					КонецПопытки;
				КонецЕсли;
			КонецЕсли;
		КонецЕсли;
		
		Если НайденыОбороты И НайденВес Тогда
			Прервать;
		КонецЕсли;
		
	КонецЦикла;
	
	Если НЕ НайденыОбороты И НЕ НайденВес Тогда
		Сообщение = "В файле не найдены обороты и вес.";
		Возврат;
	КонецЕсли;
	
	Если НЕ НайденыОбороты Тогда
		Сообщение = "В файле не найдены обороты.";
		Возврат;
	КонецЕсли;
	
	Если НЕ НайденВес Тогда
		Сообщение = "В файле не найден Consumption.";
		Возврат;
	КонецЕсли;
	
	Обороты = ОборотыИзФайла;
	Вес = Окр(ВесГраммИзФайла / 1000, 2);
	
	ДанныеСтрокиВыпуска = Неопределено;
	
	Если Элементы.Выпуск.ТекущиеДанные <> Неопределено Тогда
		ДанныеСтрокиВыпуска = Элементы.Выпуск.ТекущиеДанные;
	Иначе
		ВыдСтроки = Элементы.Выпуск.ВыделенныеСтроки;
		Для каждого СтрокаДС Из ВыдСтроки Цикл
			ДанныеСтроки = Элементы.Выпуск.ДанныеСтроки(СтрокаДС);
			Если ДанныеСтроки <> Неопределено Тогда
				ДанныеСтрокиВыпуска = ДанныеСтроки;
				Прервать;
			КонецЕсли;
		КонецЦикла;
	КонецЕсли;
	
	Если ДанныеСтрокиВыпуска = Неопределено Тогда
		Обороты = 0;
		Вес = 0;
		Длина = 0;
		МатериалыИзФайла.Очистить();
		ФидерыИзФайла.Очистить();
		Сообщение = "Сначала выберите строку выпуска. Загрузка файла станка запрещена.";
		Возврат;
	КонецЕсли;
	
	Объект.ССылкаНОМ = ДанныеСтрокиВыпуска.НоменклатураПФ;
	Объект.ССылкаНОМГП = ДанныеСтрокиВыпуска.Продукция;
	АктивныйЗаказНП = ДанныеСтрокиВыпуска.Ссылка;
	
	Плотность = ПолучитьПлотность();
	ШиринаРулона = ПолучитьШирину();
	
	Если ЗначениеЗаполнено(Плотность) И ЗначениеЗаполнено(ШиринаРулона) Тогда
		ВесПриИзменении();
	Иначе
		Длина = 0;
	КонецЕсли;
	
	КлючФайлаСтанкаВрем = СформироватьКлючФайлаСтанкаИзТекстДокумента(ТекстДок);
	СМАРТ_КлючФайлаСтанкаMayer = КлючФайлаСтанкаВрем;
	
	Если НЕ ЗначениеЗаполнено(КлючФайлаСтанкаВрем) Тогда
		Сообщение = "В файле отсутствует Product group или ключ не сформирован.";
		Возврат;
	КонецЕсли;
	
	РезультатПроверкиКлюча = ПроверитьКлючФайлаДляВыбраннойНоменклатуры(КлючФайлаСтанкаВрем);
	Если НЕ РезультатПроверкиКлюча.Разрешено Тогда
		Сообщение = РезультатПроверкиКлюча.Текст;
		Возврат;
	КонецЕсли;
	Если ЗначениеЗаполнено(РезультатПроверкиКлюча.Текст) Тогда
		ДопСообщение = РезультатПроверкиКлюча.Текст;
	КонецЕсли;
	
	Если НЕ ПодтвердитьЗагрузкуФайлаСтанка(ИмяУстройства, КлючФайлаСтанкаВрем) Тогда
		СброситьДанныеПослеОтказаПоФайлуСтанка();
		Сообщение = "Загрузка файла станка отменена пользователем.";
		Возврат;
	КонецЕсли;
	
	ИспользоватьДанныеСтанка = Истина;
	UIDФайлаСтанка = Строка(Новый УникальныйИдентификатор);
	ИмяФайлаСтанка = ПутьКФайлу;
	ИдентификаторСтанкаИзФайла = ИмяУстройства;
	ХэшФайлаСтанка = UIDФайлаСтанка;
	
	ДатаФайла = ПреобразоватьДатуLGL(ДатаОтчетаСтрокой);
	Если ДатаФайла <> Неопределено Тогда
		ДатаФайлаСтанка = ДатаФайла;
		//Объект.ДатаФайлаСтанка = ДатаФайла;
	Иначе
		ДатаФайлаСтанка = Неопределено;
	КонецЕсли;
	
	ОборотыВФайле = ОборотыИзФайла;
	ВесВФайле = Окр(ВесГраммИзФайла, 3);
	
	ПересчитатьПоляСтанка();
	ЗаполнитьМатериалыИзФайлаПоProductGroup(ТекстДок);
	ЗаполнитьФидерыИзФайла(ТекстДок);
	
	Если ФидерыИзФайла.Количество() = 0 Тогда
		Если Вопрос("В файле нет блока Feeder. Продолжить загрузку?", РежимДиалогаВопрос.ДаНет, 0) <> КодВозвратаДиалога.Да Тогда
			СброситьДанныеПослеОтказаПоФайлуСтанка();
			Сообщение = "Загрузка файла отменена.";
			Возврат;
		КонецЕсли;
	КонецЕсли;
	
	СверитьМатериалыИФидеры();
	
	Если ЕстьРасхожденияМатериалыФидеры() Тогда
		Если Вопрос("Есть заметные расхождения Product group и Feeder. Продолжить загрузку?", РежимДиалогаВопрос.ДаНет, 0) <> КодВозвратаДиалога.Да Тогда
			СброситьДанныеПослеОтказаПоФайлуСтанка();
			Сообщение = "Загрузка файла отменена.";
			Возврат;
		КонецЕсли;
	КонецЕсли;
	
	Если НЕ ПодтвердитьБольшоеОтклонениеПоВесу() Тогда
		СброситьДанныеПослеОтказаПоФайлуСтанка();
		Сообщение = "Загрузка файла станка отменена пользователем из-за большого отклонения веса.";
		Возврат;
	КонецЕсли;
	
		ПредупреждениеПоСтанку = "";
	ОжидаемыйИдентификаторСтанка = ПолучитьИдентификаторВыбранногоСтанка();
	КороткийКодФайла = ПолучитьКороткийКодСтанка(ИдентификаторСтанкаИзФайла);
	
	Если ЗначениеЗаполнено(ОжидаемыйИдентификаторСтанка)
		И ЗначениеЗаполнено(КороткийКодФайла)
		И ОжидаемыйИдентификаторСтанка <> КороткийКодФайла Тогда
		
		ПредупреждениеПоСтанку = " ВНИМАНИЕ! Выбран станок " + Строка(Станок)
		+ " (код " + ОжидаемыйИдентификаторСтанка + "), "
		+ "а файл загружен со станка " + ИдентификаторСтанкаИзФайла
		+ " (код " + КороткийКодФайла + ").";
	КонецЕсли;
	
	Сообщение = "Загружен файл станка. Обороты: "
		+ Формат(Обороты, "ЧГ=0")
		+ ", вес: " + Формат(Вес, "ЧДЦ=2") + " кг";
	
	Если ЗначениеЗаполнено(ДатаОтчетаСтрокой) Тогда
		Сообщение = Сообщение + ", отчет: " + ДатаОтчетаСтрокой;
	КонецЕсли;
	
	Если ЗначениеЗаполнено(ИмяУстройства) Тогда
		Сообщение = Сообщение + ", устройство: " + ИмяУстройства;
	КонецЕсли;
	
	Если ЗначениеЗаполнено(КлючФайлаСтанкаВрем) Тогда
		Сообщение = Сообщение + ", ключ: " + КлючФайлаСтанкаВрем;
	КонецЕсли;
	
	Если ЗначениеЗаполнено(ПредупреждениеПоСтанку) Тогда
		Сообщение = Сообщение + ПредупреждениеПоСтанку;
	КонецЕсли;
	
	КоличествоНесопоставленных = ПолучитьКоличествоНесопоставленныхНитейИзФайла();
	Если КоличествоНесопоставленных > 0 Тогда
		ПредупреждениеНесопоставленные = "Не сопоставлено нитей: " + Формат(КоличествоНесопоставленных, "ЧГ=0") + ". "
			+ "Сначала назначьте материалы для всех несопоставленных нитей.";
		Если ЗначениеЗаполнено(ДопСообщение) Тогда
			ДопСообщение = ДопСообщение + " " + ПредупреждениеНесопоставленные;
		Иначе
			ДопСообщение = ПредупреждениеНесопоставленные;
		КонецЕсли;
	КонецЕсли;
	
	ЗаписатьДанныеФайлаСтанкаВРегистрыНаСервере(
		UIDФайлаСтанка,
		ПутьКФайлу,
		ИдентификаторСтанкаИзФайла
	);
	
	Если ЗначениеЗаполнено(ДопСообщение) Тогда
		Сообщение = Сообщение + ". " + ДопСообщение;
	КонецЕсли;
	
КонецПроцедуры

&НаСервере
Функция ПолучитьКоличествоНесопоставленныхНитейИзФайлаНаСервере()
	КоличествоНесопоставленных = 0;
	Для каждого СтрокаМатериала Из МатериалыИзФайла Цикл
		Если ЗначениеЗаполнено(СтрокаМатериала.КлючНитиИзФайла)
			И (НЕ ЗначениеЗаполнено(СтрокаМатериала.НоменклатураМатериала)
				ИЛИ НЕ СтрокаМатериала.Сопоставлено) Тогда
			КоличествоНесопоставленных = КоличествоНесопоставленных + 1;
		КонецЕсли;
	КонецЦикла;
	Возврат КоличествоНесопоставленных;
КонецФункции

&НаКлиенте
Функция ПроверитьКлючФайлаДляВыбраннойНоменклатуры(КлючФайла)
	
	Результат = Новый Структура("Разрешено,Текст", Истина, "");
	
	Если НЕ ЗначениеЗаполнено(Объект.ССылкаНОМ) Тогда
		Результат.Разрешено = Ложь;
		Результат.Текст = "Не выбрана номенклатура ПФ.";
		Возврат Результат;
	КонецЕсли;
	
	КлючНоменклатуры = ПолучитьКлючФайлаСтанкаНоменклатурыНаСервере(Объект.ССылкаНОМ);
	КлючФайлаНорм = СокрЛП(Строка(КлючФайла));
	
	Если НЕ ЗначениеЗаполнено(КлючНоменклатуры) Тогда
		Результат.Текст = "Для выбранной номенклатуры ПФ ключ файла пока не настроен.";
		Возврат Результат;
	КонецЕсли;
	
	Если КлючНоменклатуры <> КлючФайлаНорм Тогда
		Результат.Разрешено = Ложь;
		Результат.Текст = "Ключ файла не соответствует выбранной номенклатуре ПФ.";
	КонецЕсли;
	
	Возврат Результат;
	
КонецФункции

&НаКлиенте
Процедура ЗаписатьКлючФайлаВСправочник(Команда)
	
	Если НЕ ЗначениеЗаполнено(Объект.ССылкаНОМ) Тогда
		Сообщение = "Сначала выберите номенклатуру ПФ.";
		Возврат;
	КонецЕсли;
	
	КлючФайла = СокрЛП(Строка(СМАРТ_КлючФайлаСтанкаMayer));
	Если НЕ ЗначениеЗаполнено(КлючФайла) Тогда
		Сообщение = "Сначала загрузите файл станка и сформируйте ключ.";
		Возврат;
	КонецЕсли;
	
	ТекущийКлюч = ПолучитьКлючФайлаСтанкаНоменклатурыНаСервере(Объект.ССылкаНОМ);
	Если ЗначениеЗаполнено(ТекущийКлюч) Тогда
		Если ТекущийКлюч = КлючФайла Тогда
			Сообщение = "Ключ уже установлен у выбранной номенклатуры.";
		Иначе
			Сообщение = "У выбранной номенклатуры уже записан другой ключ. Перезапись запрещена.";
		КонецЕсли;
		Возврат;
	КонецЕсли;
	
	Дубликат = НайтиНоменклатуруПоКлючуФайлаСтанка(КлючФайла, Объект.ССылкаНОМ);
	Если ЗначениеЗаполнено(Дубликат) Тогда
		Сообщение = "Этот ключ уже используется номенклатурой: " + Строка(Дубликат) + ".";
		Возврат;
	КонецЕсли;
	
	ЗаписатьКлючФайлаВСправочникНаСервере(Объект.ССылкаНОМ, КлючФайла);
	Сообщение = "Ключ файла записан в номенклатуру ПФ.";
	
КонецПроцедуры

&НаСервере
Функция НайтиНоменклатуруПоКлючуФайлаСтанка(КлючФайла, ИсключитьНоменклатуру)
	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ ПЕРВЫЕ 1
	|	Номенклатура.Ссылка КАК Ссылка
	|ИЗ
	|	Справочник.Номенклатура КАК Номенклатура
	|ГДЕ
	|	Номенклатура.СМАРТ_КлючФайлаСтанкаMayer = &КлючФайла
	|	И Номенклатура.Ссылка <> &ИсключитьНоменклатуру";
	Запрос.УстановитьПараметр("КлючФайла", КлючФайла);
	Запрос.УстановитьПараметр("ИсключитьНоменклатуру", ИсключитьНоменклатуру);
	Выборка = Запрос.Выполнить().Выбрать();
	Если Выборка.Следующий() Тогда
		Возврат Выборка.Ссылка;
	КонецЕсли;
	Возврат Справочники.Номенклатура.ПустаяСсылка();
КонецФункции

Процедура ЗаписатьДанныеФайлаСтанкаВРегистрыНаСервере(UIDФайла, ПутьКФайлу, ИдентификаторСтанка)
	
	Если НЕ ИспользоватьДанныеСтанка Тогда
		Возврат;
	КонецЕсли;
	
	НаборФайлы = РегистрыСведений.кмФайлыСтанкаMayer.СоздатьНаборЗаписей();
	ЗаписьФайла = НаборФайлы.Добавить();
	ЗаписьФайла.UIDФайла = UIDФайла;
	ЗаписьФайла.ИмяФайла = ИмяФайлаСтанка;
	ЗаписьФайла.Номенклатура = Объект.ССылкаНОМ;
	ЗаписьФайла.Станок = Станок;
	ЗаписьФайла.ДатаФайла = ДатаФайлаСтанка;
	ЗаписьФайла.РежимЗагрузки = "ЗагрузкаФайла";
	ЗаписьФайла.ОборотыВФайле = ОборотыВФайле;
	ЗаписьФайла.ОбщийВесВФайле = ВесВФайле;

		Возврат 0;
	КонецЕсли;
	
	СтрЧ = СтрЗаменить(СтрЧ, " ", "");
	ПозицияE = СтрНайти(ВРег(СтрЧ), "E");

		КонецЕсли;

		ПорядокСтр = Сред(СтрЧ, ПозицияE + 1);
		
		Мантисса = Число(СтрЗаменить(МантиссаСтр, ".", ","));
		ПорядокСтр = СтрЗаменить(ПорядокСтр, "+", "");
		Порядок = Число(ПорядокСтр);
		
		Результат = Мантисса;
		
		Если Порядок > 0 Тогда
			Для Инд = 1 По Порядок Цикл
				Результат = Результат * 10;
			КонецЦикла;
		ИначеЕсли Порядок < 0 Тогда
			Для Инд = 1 По -Порядок Цикл
				Результат = Результат / 10;
			КонецЦикла;
		КонецЕсли;
		
		Возврат Результат;
		
	Исключение
		Возврат 0;
	КонецПопытки;

Функция ЭтоСтрокаProductGroupLGL(СтрокаФайла)
	
	НормСтрока = НормализоватьСтрокуLGL(СтрокаФайла);
	
	Если НормСтрока = "" Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Если НормСтрока = "Product group" Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Если СтрНайти(ВРег(НормСтрока), "FEEDER") > 0 Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Части = СтрРазделить(НормСтрока, ",");
	
	Если Части.Количество() < 11 Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Если ВРег(СокрЛП(Части[3])) = "CM"
		И ВРег(СокрЛП(Части[5])) = "%CM"
		И ВРег(СокрЛП(Части[7])) = "G"
		И ВРег(СокрЛП(Части[9])) = "%G" Тогда
		Возврат Истина;
	КонецЕсли;
	
	Возврат Ложь;

&НаКлиенте
Процедура ЗаполнитьМатериалыИзФайлаПоProductGroup(ТекстДок)
	
	МатериалыИзФайла.Очистить();
	
	НачалсяБлокProductGroup = Ложь;
	
	Для НомерСтроки = 1 По ТекстДок.КоличествоСтрок() Цикл
		
		СтрокаФайла = ТекстДок.ПолучитьСтроку(НомерСтроки);
		СтрокаФайла = СтрЗаменить(СтрокаФайла, Символы.ВК, "");
		СтрокаФайла = СтрЗаменить(СтрокаФайла, Символы.ПС, "");
		НормСтрока = НормализоватьСтрокуLGL(СтрокаФайла);
		
		Если НормСтрока = "Product group" Тогда
			НачалсяБлокProductGroup = Истина;
			Продолжить;
		КонецЕсли;
		
		Если НЕ НачалсяБлокProductGroup Тогда
			Продолжить;
		КонецЕсли;
		
		Если СтрНайти(ВРег(НормСтрока), "FEEDER") > 0 Тогда
			Прервать;
		КонецЕсли;
		
		Если НЕ ЭтоСтрокаProductGroupLGL(НормСтрока) Тогда
			Продолжить;
		КонецЕсли;
		
		Части = СтрРазделить(НормСтрока, ",");
		
		НоваяСтрока = МатериалыИзФайла.Добавить();
		НоваяСтрока.НомерСтр = МатериалыИзФайла.Количество();
		
		// В файле:
		// [0] = Count/тип вроде 25 Tex, 167 dTex
		// [1] = имя/обозначение нити вроде milang, 3603, трилобал
		НоваяСтрока.ТипНити = СокрЛП(Части[0]);

&НаКлиенте
Процедура ПересчитатьПоляСтанка()
	
	// 1. Показатели файла станка
	
	Если ЗначениеЗаполнено(ВесВФайле)
		И ЗначениеЗаполнено(Плотность)
		И ЗначениеЗаполнено(ШиринаРулона)
		И Плотность <> 0
		И ШиринаРулона <> 0 Тогда
		
		// длина файла в сантиметрах
		ДлинаСмВФайле = Окр(ВесВФайле * 10000 / (Плотность * ШиринаРулона), 6);
		
	Иначе
		ДлинаСмВФайле = 0;
	КонецЕсли;
	
	Если ЗначениеЗаполнено(ОборотыВФайле) И ОборотыВФайле <> 0 Тогда
		
		Если ЗначениеЗаполнено(ДлинаСмВФайле) Тогда
			ДлинаНа1ОборотСм = Окр(ДлинаСмВФайле / ОборотыВФайле, 6);
		Иначе
			ДлинаНа1ОборотСм = 0;
		КонецЕсли;
		
		Если ЗначениеЗаполнено(ВесВФайле) Тогда
			ВесНа1ОборотГ = Окр(ВесВФайле / ОборотыВФайле, 6);
		Иначе
			ВесНа1ОборотГ = 0;
		КонецЕсли;
		
	Иначе
		ДлинаНа1ОборотСм = 0;
		ВесНа1ОборотГ = 0;
	КонецЕсли;
	
	// пока коэффициент всегда 1
	Если ЗначениеЗаполнено(ВесНа1ОборотГ) И ВесНа1ОборотГ <> 0 Тогда
		КоэффициентПересчетаФайла = 1;
	Иначе
		КоэффициентПересчетаФайла = 0;
	КонецЕсли;
	
	
	// 2. Вес расчетный по файлу
	
	Если ЗначениеЗаполнено(Обороты)
		И ЗначениеЗаполнено(ВесНа1ОборотГ)
		И ЗначениеЗаполнено(КоэффициентПересчетаФайла)
		И ВесНа1ОборотГ <> 0
		И КоэффициентПересчетаФайла <> 0 Тогда
		
		ВесРасчетныйПоФайлу = Окр(Обороты * ВесНа1ОборотГ * КоэффициентПересчетаФайла / 1000, 6); // кг
		
	Иначе
		ВесРасчетныйПоФайлу = 0;
	КонецЕсли;
	
	
	// 3. Длина расчетная:
	// если есть файл и обороты — считаем по файлу;
	// если файла нет — запасной вариант по старой формуле
	
	Если ЗначениеЗаполнено(Обороты)
		И ЗначениеЗаполнено(ДлинаНа1ОборотСм)
		И ЗначениеЗаполнено(КоэффициентПересчетаФайла)
		И ДлинаНа1ОборотСм <> 0
		И КоэффициентПересчетаФайла <> 0 Тогда
		
		ДлинаРасчетнаяМ = Окр(Обороты * ДлинаНа1ОборотСм * КоэффициентПересчетаФайла / 100, 3);
		
	ИначеЕсли ЗначениеЗаполнено(Вес)
		И ЗначениеЗаполнено(Плотность)
		И ЗначениеЗаполнено(ШиринаРулона)
		И Плотность <> 0
		И ШиринаРулона <> 0 Тогда
		
		ДлинаРасчетнаяМ = Окр(Вес * 1000 / (Плотность * ШиринаРулона / 100), 3);
		
	Иначе
		ДлинаРасчетнаяМ = 0;
	КонецЕсли;
	
	
	// 4. Плотность расчетная:
	// если есть длина расчетная по файлу — считаем от нее,
	// иначе от фактической длины
	
	Если ЗначениеЗаполнено(Вес)
		И ЗначениеЗаполнено(ШиринаРулона)
		И ШиринаРулона <> 0 Тогда
		
		Если ЗначениеЗаполнено(ДлинаРасчетнаяМ) И ДлинаРасчетнаяМ <> 0 Тогда
			ПлотностьРасчетная = Окр(Вес * 100000 / (ДлинаРасчетнаяМ * ШиринаРулона), 3);
		ИначеЕсли ЗначениеЗаполнено(Длина) И Длина <> 0 Тогда
			ПлотностьРасчетная = Окр(Вес * 100000 / (Длина * ШиринаРулона), 3);
		Иначе
			ПлотностьРасчетная = 0;
		КонецЕсли;
		
	Иначе
		ПлотностьРасчетная = 0;
	КонецЕсли;
	
	
	// 5. Отклонение веса
	
	Если ЗначениеЗаполнено(ВесРасчетныйПоФайлу)
		И ВесРасчетныйПоФайлу <> 0
		И ЗначениеЗаполнено(Вес) Тогда
		
		ОтклонениеВесПроц = Окр((Вес - ВесРасчетныйПоФайлу) * 100 / ВесРасчетныйПоФайлу, 3);
		
	Иначе
		ОтклонениеВесПроц = 0;
	КонецЕсли;
	
КонецПроцедуры

Функция ЭтоСтрокаFeederLGL(СтрокаФайла)
	
	НормСтрока = НормализоватьСтрокуLGL(СтрокаФайла);
	
	Если НормСтрока = "" Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Если СтрНайти(ВРег(НормСтрока), "FEEDER") > 0 Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Части = СтрРазделить(НормСтрока, ",");
	
	Если Части.Количество() < 11 Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Если СокрЛП(Части[0]) <> "" Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Если ВРег(СокрЛП(Части[3])) = "CM"
		И ВРег(СокрЛП(Части[5])) = "G"
		И ВРег(СокрЛП(Части[7])) = "CM" Тогда
		Возврат Истина;
	КонецЕсли;
	
	Возврат Ложь;

Процедура СверитьМатериалыИФидеры()
	
	Если МатериалыИзФайла.Количество() = 0 Тогда
		Возврат;
	КонецЕсли;
	
	// Допуски для сверки.
	// Небольшое расхождение допустимо из-за округлений в Product group и Feeder.
	ДопускСмАбс = 10;
	ДопускСмПроц = 0.1;
	ДопускГраммАбс = 0.5;
	ДопускГраммПроц = 1;
	
	СуммыФидеров = Новый Соответствие;
	СтатусыПоКлючу = Новый Соответствие;
	
	// 1. Накопим суммы по фидерам в разрезе нити
	Для каждого СтрокаФидера Из ФидерыИзФайла Цикл
		
		КлючНити = ПолучитьКлючНитиИзФайла(СтрокаФидера.ТипНити, СтрокаФидера.НаименованиеНитиИзФайла);
		
		Итоги = СуммыФидеров.Получить(КлючНити);
		Если Итоги = Неопределено Тогда
			Итоги = Новый Структура("См,Грамм", 0, 0);
			СуммыФидеров.Вставить(КлючНити, Итоги);
		КонецЕсли; 
		
		Итоги.См = Итоги.См + СтрокаФидера.РасходСм;
		Итоги.Грамм = Итоги.Грамм + СтрокаФидера.РасходГрамм;
		
	КонецЦикла;
	
	// 2. Сверим Product group с суммами фидеров
	Для каждого СтрокаМатериала Из МатериалыИзФайла Цикл
		
		КлючНити = ПолучитьКлючНитиИзФайла(СтрокаМатериала.ТипНити, СтрокаМатериала.НаименованиеНитиИзФайла);
		
		СтрокаМатериала.РасходСмПоФидерам = 0;
		СтрокаМатериала.РасходГраммПоФидерам = 0;
		СтрокаМатериала.ОтклонениеСмПроц = 0;
		СтрокаМатериала.ОтклонениеГраммПроц = 0;
		СтрокаМатериала.ЕстьРасхождение = Ложь;
		СтрокаМатериала.СтатусСверки = "";
		
		Итоги = СуммыФидеров.Получить(КлючНити);
		
		Если Итоги <> Неопределено Тогда
			
			СтрокаМатериала.РасходСмПоФидерам = Окр(Итоги.См, 3);
			СтрокаМатериала.РасходГраммПоФидерам = Окр(Итоги.Грамм, 3);
			
			Если ЗначениеЗаполнено(СтрокаМатериала.РасходСм) И СтрокаМатериала.РасходСм <> 0 Тогда
				СтрокаМатериала.ОтклонениеСмПроц = Окр((СтрокаМатериала.РасходСмПоФидерам - СтрокаМатериала.РасходСм) * 100 / СтрокаМатериала.РасходСм, 3);
			КонецЕсли;
			
			Если ЗначениеЗаполнено(СтрокаМатериала.РасходГрамм) И СтрокаМатериала.РасходГрамм <> 0 Тогда
				СтрокаМатериала.ОтклонениеГраммПроц = Окр((СтрокаМатериала.РасходГраммПоФидерам - СтрокаМатериала.РасходГрамм) * 100 / СтрокаМатериала.РасходГрамм, 3);
			КонецЕсли;
			
			АбсОтклонениеСм = АбсЗначение(СтрокаМатериала.РасходСмПоФидерам - СтрокаМатериала.РасходСм);
			АбсОтклонениеГрамм = АбсЗначение(СтрокаМатериала.РасходГраммПоФидерам - СтрокаМатериала.РасходГрамм);
			
			Если (АбсОтклонениеСм > ДопускСмАбс И АбсЗначение(СтрокаМатериала.ОтклонениеСмПроц) > ДопускСмПроц)
				ИЛИ (АбсОтклонениеГрамм > ДопускГраммАбс И АбсЗначение(СтрокаМатериала.ОтклонениеГраммПроц) > ДопускГраммПроц) Тогда
				
				СтрокаМатериала.ЕстьРасхождение = Истина;
				СтрокаМатериала.СтатусСверки = "РАСХОЖДЕНИЕ";
				
			Иначе
				
				СтрокаМатериала.ЕстьРасхождение = Ложь;
				СтрокаМатериала.СтатусСверки = "ОК";
				
			КонецЕсли;
			
		Иначе
			
			СтрокаМатериала.ЕстьРасхождение = Истина;
			СтрокаМатериала.СтатусСверки = "НЕТ ФИДЕРОВ";
			
		КонецЕсли;
		
		СтатусыПоКлючу.Вставить(КлючНити, Новый Структура("ЕстьРасхождение,СтатусСверки", СтрокаМатериала.ЕстьРасхождение, СтрокаМатериала.СтатусСверки));
		
	КонецЦикла;
	
	// 3. Проставим статус в строки фидеров
	Для каждого СтрокаФидера Из ФидерыИзФайла Цикл
		
		КлючНити = ПолучитьКлючНитиИзФайла(СтрокаФидера.ТипНити, СтрокаФидера.НаименованиеНитиИзФайла);
		
		Статус = СтатусыПоКлючу.Получить(КлючНити);
		
		Если Статус <> Неопределено Тогда
			СтрокаФидера.ЕстьРасхождение = Статус.ЕстьРасхождение;
			СтрокаФидера.СтатусСверки = Статус.СтатусСверки;
		Иначе
			СтрокаФидера.ЕстьРасхождение = Истина;
			СтрокаФидера.СтатусСверки = "НЕТ ГРУППЫ";
		КонецЕсли;
		
	КонецЦикла;
	
КонецПроцедуры

Функция ИдентификаторСтанкаЕстьВСпискеВыбора(ИдентификаторСтанка)
	
	Если НЕ ЗначениеЗаполнено(ИдентификаторСтанка) Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Если Элементы.Станок.СписокВыбора = Неопределено Тогда
		Возврат Ложь;
	КонецЕсли;
	
	КороткийКодФайла = ПолучитьКороткийКодСтанка(ИдентификаторСтанка);
	
	Если НЕ ЗначениеЗаполнено(КороткийКодФайла) Тогда
		Возврат Ложь;
	КонецЕсли;
	
	Для каждого СтрокаСписка Из Элементы.Станок.СписокВыбора Цикл
		
		Если ПолучитьКороткийКодСтанка(СтрокаСписка.Представление) = КороткийКодФайла Тогда
			Возврат Истина;
		КонецЕсли;
		
	КонецЦикла;
	
	Возврат Ложь;

Функция ПодтвердитьЗагрузкуФайлаСтанка(ИдентификаторСтанкаИзФайла, КлючФайлаСтанкаВрем)
	
	ОжидаемыйИдентификаторСтанка = ПолучитьИдентификаторВыбранногоСтанка();
	КороткийКодФайла = ПолучитьКороткийКодСтанка(ИдентификаторСтанкаИзФайла);
	
	Если НЕ ЗначениеЗаполнено(КлючФайлаСтанкаВрем) Тогда
		
		Ответ = Вопрос(
			"Не удалось определить ключ файла станка из блока Product group." + Символы.ПС
			+ "Загрузить файл без проверки рецепта?",
			РежимДиалогаВопрос.ДаНет,
			0
		);
		
		Возврат Ответ = КодВозвратаДиалога.Да;
		
	КонецЕсли;
	
	Если НЕ ЗначениеЗаполнено(ИдентификаторСтанкаИзФайла) Тогда
		
		Ответ = Вопрос(
			"Не удалось определить идентификатор станка из файла." + Символы.ПС
			+ "Ключ файла: " + КлючФайлаСтанкаВрем + Символы.ПС
			+ "Загрузить файл без проверки станка?",
			РежимДиалогаВопрос.ДаНет,
			0
		);
		
		Возврат Ответ = КодВозвратаДиалога.Да;
		
	КонецЕсли;
	
	Если НЕ ИдентификаторСтанкаЕстьВСпискеВыбора(ИдентификаторСтанкаИзФайла) Тогда
		
		Ответ = Вопрос(
			"Станок из файла не найден в настройке формы: " + ИдентификаторСтанкаИзФайла
			+ " (код " + КороткийКодФайла + ")" + Символы.ПС
			+ "Ключ файла: " + КлючФайлаСтанкаВрем + Символы.ПС
			+ "Загрузить файл?",
			РежимДиалогаВопрос.ДаНет,
			0
		);
		
		Возврат Ответ = КодВозвратаДиалога.Да;
		
	КонецЕсли;
	
	Если ЗначениеЗаполнено(ОжидаемыйИдентификаторСтанка)
		И ЗначениеЗаполнено(КороткийКодФайла)
		И ОжидаемыйИдентификаторСтанка <> КороткийКодФайла Тогда
		
		Ответ = Вопрос(
			"Выбран станок " + Строка(Станок) + " (код " + ОжидаемыйИдентификаторСтанка + "), "
			+ "а файл загружен со станка " + ИдентификаторСтанкаИзФайла
			+ " (код " + КороткийКодФайла + ")." + Символы.ПС
			+ "Ключ файла: " + КлючФайлаСтанкаВрем + Символы.ПС
			+ "Все равно загрузить файл?",
			РежимДиалогаВопрос.ДаНет,
			0
		);
		
		Возврат Ответ = КодВозвратаДиалога.Да;
		
	КонецЕсли;
	
	Возврат Истина;

Функция НормализоватьФрагментКлючаФайлаСтанка(СтрокаИсходная)
	
	Результат = Строка(СтрокаИсходная);
	Результат = СтрЗаменить(Результат, Символы.Таб, " ");
	Результат = СтрЗаменить(Результат, Символы.ВК, "");
	Результат = СтрЗаменить(Результат, Символы.ПС, "");
	Результат = СтрЗаменить(Результат, "Ё", "Е");
	Результат = СтрЗаменить(Результат, ";", "/");
	Результат = СтрЗаменить(Результат, "|", "/");
	
	Пока СтрНайти(Результат, "  ") > 0 Цикл
		Результат = СтрЗаменить(Результат, "  ", " ");
	КонецЦикла;
	
	Результат = СокрЛП(Результат);
	Результат = ВРег(Результат);
	
	Возврат Результат;

Функция ДобавитьЭлементВКлючФайлаСтанка(КлючБезПрефикса, ЭлементКлюча)
	
	ЭлементКлюча = СокрЛП(Строка(ЭлементКлюча));
	КлючБезПрефикса = СокрЛП(Строка(КлючБезПрефикса));
	
	Если НЕ ЗначениеЗаполнено(ЭлементКлюча) Тогда
		Возврат КлючБезПрефикса;
	КонецЕсли;
	
	Если НЕ ЗначениеЗаполнено(КлючБезПрефикса) Тогда
		Возврат ЭлементКлюча;
	КонецЕсли;
	
	// Уже есть такой элемент
	Если СтрНайти(";" + КлючБезПрефикса + ";", ";" + ЭлементКлюча + ";") > 0 Тогда
		Возврат КлючБезПрефикса;
	КонецЕсли;
	
	Части = СтрРазделить(КлючБезПрефикса, ";");
	НовыйКлюч = "";
	Вставлен = Ложь;
	
	Для каждого ТекЭлемент Из Части Цикл
		
		ТекЭлемент = СокрЛП(Строка(ТекЭлемент));
		Если НЕ ЗначениеЗаполнено(ТекЭлемент) Тогда
			Продолжить;
		КонецЕсли;
		
		Если НЕ Вставлен И ЭлементКлюча < ТекЭлемент Тогда
			Если ЗначениеЗаполнено(НовыйКлюч) Тогда
				НовыйКлюч = НовыйКлюч + ";";
			КонецЕсли;
			НовыйКлюч = НовыйКлюч + ЭлементКлюча;
			Вставлен = Истина;
		КонецЕсли;
		
		Если ЗначениеЗаполнено(НовыйКлюч) Тогда
			НовыйКлюч = НовыйКлюч + ";";
		КонецЕсли;
		НовыйКлюч = НовыйКлюч + ТекЭлемент;
		
	КонецЦикла;
	
	Если НЕ Вставлен Тогда
		Если ЗначениеЗаполнено(НовыйКлюч) Тогда
			НовыйКлюч = НовыйКлюч + ";";
		КонецЕсли;
		НовыйКлюч = НовыйКлюч + ЭлементКлюча;
	КонецЕсли;
	
	Возврат НовыйКлюч;
	
КонецФункции

Функция СформироватьКлючФайлаСтанкаИзТекстДокумента(ТекстДок)
	
	КлючБезПрефикса = "";
	НачалсяБлокProductGroup = Ложь;
	
	Для НомерСтроки = 1 По ТекстДок.КоличествоСтрок() Цикл
		
		СтрокаФайла = ТекстДок.ПолучитьСтроку(НомерСтроки);
		СтрокаФайла = СтрЗаменить(СтрокаФайла, Символы.ВК, "");
		СтрокаФайла = СтрЗаменить(СтрокаФайла, Символы.ПС, "");
		НормСтрока = НормализоватьСтрокуLGL(СтрокаФайла);
		
		Если ВРег(НормСтрока) = "PRODUCT GROUP" Тогда
			НачалсяБлокProductGroup = Истина;
			Продолжить;
		КонецЕсли;
		
		Если НЕ НачалсяБлокProductGroup Тогда
			Продолжить;
		КонецЕсли;
		
		Если СтрНайти(ВРег(НормСтрока), "FEEDER") > 0 Тогда
			Прервать;
		КонецЕсли;
		
		Если НЕ ЭтоСтрокаProductGroupLGL(НормСтрока) Тогда
			Продолжить;
		КонецЕсли;
		
		Части = СтрРазделить(НормСтрока, ",");
		Если Части.Количество() < 2 Тогда
			Продолжить;
		КонецЕсли;
		
		ЭлементКлюча = СформироватьЭлементКлючаФайлаСтанка(Части[0], Части[1]);
		
		Если ЗначениеЗаполнено(ЭлементКлюча) Тогда
			КлючБезПрефикса = ДобавитьЭлементВКлючФайлаСтанка(КлючБезПрефикса, ЭлементКлюча);
		КонецЕсли;
		
	КонецЦикла;
	
	Если НЕ ЗначениеЗаполнено(КлючБезПрефикса) Тогда
		Возврат "";
	КонецЕсли;
	
	Возврат "PG:" + КлючБезПрефикса;
	
КонецФункции

Функция ПолучитьКороткийКодСтанка(ИдентификаторСтанка)
	
	СтрКод = СокрЛП(Строка(ИдентификаторСтанка));
	
	Если НЕ ЗначениеЗаполнено(СтрКод) Тогда
		Возврат "";
	КонецЕсли;
	
	СтрКод = ВРег(СтрКод);
	
	Если СтрДлина(СтрКод) >= 4 Тогда
